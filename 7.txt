p([],_,_,[]):- %пустой список
    !.
p([X|St],N,K,[X1|F]):- %запись значений, когда счетчик равен номеру указанного элемента
	K>=N,
    K1 is K+1,
    X1 is X,
    p(St,N,K1,F),!.
 p([_|St],N,K,F):- %пропуск, если счетчик меньше указанного значения 
    K<N,
    K1 is K+1,
    p(St,N,K1,F),!.
